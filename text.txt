// Version Control using Git and Command line:

mkdir Story
cd story

touch text.txt
code text.txt

touch ch1.txt
code ch1.txt

git init

ls
ls -a

git status

git add ch1.txt

git add .

git commit -m "Initial commit"

git log

git commit -m "Complete chapter 1"

touch ch2.txt
code ch2.txt

touch ch3.txt
code ch3.txt

git add .

git commit -m "Complete chapter 2 and 3"

git diff ch3.txt

git checkout ch3.txt

git add .

git commit -m "Done video 1"


// Github and remote repositories:

git remote add origin https://github.com/vibha-cloud/Version-Control-using-Github.git

git branch -M main

git push -u origin main

# changes performed

git add .

git commit -m "Done video 2"

git status

git push


// gitignore:

touch secrets.txt
code secrets.txt

touch .gitignore

ls -a

git add .

git status

git rm --cached -r . (staging area -> unstaged area)

git status

code .gitignore (add file names to be ignored)

git add .

git commit -m "Done video 3"

git push
